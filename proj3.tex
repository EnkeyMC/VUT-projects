	\section{Tabulky}
	Pro sázení tabulek můžeme použít buď prostředí \texttt{tabbing} nebo prostředí \texttt{tabular}.

	\subsection{Prostředí \texttt{tabbing}}
	Při použití \texttt{tabbing} vypadá tabulka následovně:
	
		\begin{tabbing}
			Vodní melouny\qquad \= Cena \qquad \= \kill
			\textbf{Ovoce} 	\> \textbf{Cena} 	\> \textbf{Množství} \\
			Jablka 			\> 25,90			\> 3 kg \\
			Hruška			\> 27,40			\> 2,5 kg \\
			Vodní melouny	\> 35,--				\> 1 kus \\
		\end{tabbing}
	
	\noindent Toto prostředí se dá také použít pro sázení algoritmů, ovšem vhodnější je použít prostředí \texttt{algorithm} nebo \texttt{algorithm2e} (viz sekce \ref{algoritmy}).

	\section{Algoritmy} \label{algoritmy}
	Pokud budeme chtít vysázet algoritmus, můžeme použít prostředí \texttt{algorithm}\footnote{Pro nápovědu, jak zacházet s prostředím \texttt{algorithm}, můžeme zkusit tuhle stránku:\newline http://ftp.cstug.cz/pub/tex/CTAN/macros/latex/contrib/algorithms/algorithms.pdf.} nebo \texttt{algorithm2e}\footnote{Pro \texttt{algorithm2e} zase tuhle: http://ftp.cstug.cz/pub/tex/CTAN/macros/latex/contrib/algorithm2e/algorithm2e.pdf.}. Příklad použití prostředí \texttt{algorithm2e} viz Algoritmus .... \\[1em]
	
	\IncMargin{1.5em}
	{\SetAlgoNoLine
	\begin{algorithm}[H]
		\caption{\textsc{FastSLAM}}
		\SetNlSty{}{}{:}
		\SetAlgoNlRelativeSize{-1}
		\SetKwFor{For}{for}{do}{endfor}
				
		\Indm
		\KwIn{$(X_{t-1}, u_t, z_t)$}
		\KwOut{$X_t$}
		\BlankLine
		\Indp
		
		$\overline{X_t} = X_t = 0$ \\
		\For{$k$ \upshape = 1 to $M$}{
			$x_t^{[k]} = \mathit{sample\_motion\_model}(u_t, x_{t-1}^{[k]})$ \\
			$\omega_t^{[k]} = \mathit{measurement\_model}(z_t, x_t^{[k]}, m_{t-1})$ \\
			$m_t^{[k]} = \mathit{updated\_occupancy\_grid}(z_t, x_t^{[k]}, m_{t-1}^{[k]})$ \\
			$\overline{X_t} = \overline{X_t} + \big<x_x^{[k]},m_t^{[k]}\big>$
		}
	
		\For{$k$ \upshape = 1 to $M$}{
			draw $i$ with probability $\approx \omega_t^{[i]}$ \\
			add $\big<x_x^{[k]}, m_t^{[k]}\big>$ to $X_t$
		}
		\KwRet{$X_t$}
	\end{algorithm}
	}
